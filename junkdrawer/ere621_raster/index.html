trees <- read.csv("~/data/ifs_all_data.csv")
trees <- trees[trees$CODE85 == 1, ]
trees$NORTH <- as.numeric(trees$NORTH)
trees$EAST <- as.numeric(trees$EAST)
trees <- sf::st_as_sf(trees, coords = c("EAST", "NORTH"))
library(ggplot2)
ggplot(trees) + 
  geom_sf()
